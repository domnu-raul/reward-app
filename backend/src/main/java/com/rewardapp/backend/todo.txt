DB STRUCTURE:
    USERS: // exists
        id - serial pk
        username - varchar(80)
        email - varchar(255)
        verified - bool - false
        register_date - date - current_date
        type - user_type - 'user'

        INTERNAL_USERS:
            password - varchar(255);

    SESSIONS: // exists
        id - serial pk
        session_id - varchar(255)
        exp_date - date - now() + 3 days
        user_id - int fk users(id)

    EMAIL_TOKENS: // exists
        id - serial pk
        token - varchar(255) not null
        exp_date - date - now() + 30 days
        user_id - int fk users(id)

    RECYCLING_CENTERS: // exists
        id - serial pk
        name - varchar(255)
        materials - material_type[]
        recyclingCenterLocation - location_type
        starting_time -  TIME
        end_time - TIME

    CONTRIBUTIONS: // exists
        id - int pk
        user_id - int fk users(id)
        recycling_center_id - int fk recycling_centers(id)
        timestamp-  TIMESTAMP - now()
        material - material
        quantity - INT
        measurment - measurment?

    VOUCHERS:
        id - serial pk
        user_id - int fk users(id)
        option_id - int fk purchase_options(id);
        code - varchar(80) -- uuid;

    PURCHASE_OPTIONS:
        id - int pk
        value - int
        cost - int
        currency - currency_type

    CUSTOM_TYPES:
        ENUM user_type ('admin', 'user')
        ENUM material_type
            'PLASTIC',
             'METAL',
             'PAPER',
             'GLASS',
             'ALUMINUM_CANS',
             'E_WASTE'

        location_type
            county VARCHAR(50),
            city VARCHAR(50),
            address VARCHAR(255),
            postal_code VARCHAR(20),

            coordinates geography(Point, 4326)
